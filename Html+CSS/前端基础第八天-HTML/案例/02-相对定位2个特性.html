package com.entity; import java.io.Serializable; import java.util.Objects; public class CarVo implements Serializable { private int Cno;//驾驶证号 private String Cnumber;//车牌号 private String Cpname;//车主信息（车主姓名） //get、set方法 public int getCno() { return Cno; } public void setCno(int cno) { Cno = cno; } public String getCnumber() { return Cnumber; } public void setCnumber(String cnumber) { Cnumber = cnumber; } public String getCpname() { return Cpname; } public void setCpname(String cpname) { Cpname = cpname; } //构造器 public CarVo(){}; public CarVo(int cno, String cnumber, String cpname) { Cno = cno; Cnumber = cnumber; Cpname = cpname; } //将equals表示相等 @Override public boolean equals(Object o) { if (this == o) return true; if (!(o instanceof CarVo)) return false; CarVo carVo = (CarVo) o; return getCno() == carVo.getCno() && getCnumber() == carVo.getCnumber() && getCpname() == carVo.getCpname(); } @Override public int hashCode() { return Objects.hash(getCno(), getCnumber(), getCpname()); } //重写toString方法 @Override public String toString() { return "CarVo{" + "Cno=" + Cno + ", Cnumber=" + Cnumber + ", Cpname=" + Cpname + '}'; } }
